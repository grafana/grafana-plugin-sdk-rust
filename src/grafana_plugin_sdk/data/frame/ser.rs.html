26impl Serialize for Frame {
27    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
28    where
29        S: Serializer,
30    {
31        let schema_fields: Vec<_> = self
32            .fields
33            .iter()
34            .map(|f| {
35                Ok(SerializableField {
36                    name: f.name.as_str(),
37                    labels: &f.labels,
38                    config: f.config.as_ref(),
39                    type_: f.type_info.simple_type(),
40                    type_info: &f.type_info,
41                })
42            })
43            .collect::<Result<_, _>>()?;
44        let ser = SerializableFrame {
45            schema: Some(SerializableFrameSchema {
46                name: &self.name,
47                ref_id: self.ref_id.as_deref(),
48                meta: &self.meta,
49                fields: &schema_fields,
50            }),
51            data: Some(SerializableFrameData {
52                fields: &self.fields,
53            }),
54        };
55        ser.serialize(serializer)
56    }
57}

93impl<'a> Serialize for SerializableFrameData<'a> {
94    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
95    where
96        S: Serializer,
97    {
98        let n_fields = self.fields.len();
99
100        let entities: Vec<_> = std::iter::repeat_n(RefCell::new(None), n_fields).collect();
101
102        let values = SerializableFrameDataValues {
103            fields: self.fields,
104            entities: &entities,
105        };
106
107        let mut map = serializer.serialize_map(None)?;
108        map.serialize_key("values")?;
109        map.serialize_value(&values)?;
110
111        if entities.iter().any(|r| r.borrow().is_some()) {
112            map.serialize_key("entities")?;
113            map.serialize_value(&entities)?;
114        }
115
116        map.end()
117    }
118}

120struct SerializableFrameDataValues<'a, 'b> {
121    fields: &'a [Field],
122    entities: &'b [RefCell<Option<Entities>>],
123}
124
125impl<'a, 'b> Serialize for SerializableFrameDataValues<'a, 'b> {
126    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
127    where
128        S: Serializer,
129    {
130        let mut seq = serializer.serialize_seq(Some(self.fields.len()))?;
131        for (arr, e) in self.fields.iter().zip(self.entities.iter()) {
132            seq.serialize_element(&SerializableArray(&*arr.values, e))?;
133        }
134        seq.end()
135    }
136}

138struct SerializableArray<'a>(&'a dyn Array, &'a RefCell<Option<Entities>>);
139
140impl<'a> Serialize for SerializableArray<'a> {
141    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
142    where
143        S: Serializer,
144    {
145        let array = self.0;
146        let len = array.len();
147        match array.data_type() {
148            DataType::Null => serializer.collect_seq(std::iter::repeat_n::<Option<()>>(None, len)),
149            DataType::Boolean => serializer.collect_seq(
150                array
151                    .as_any()
152                    .downcast_ref::<BooleanArray>()
153                    .unwrap()
154                    .iter(),
155            ),
156            DataType::Utf8 | DataType::LargeUtf8 => {
157                serializer.collect_seq(array.as_any().downcast_ref::<StringArray>().unwrap().iter())
158            }
159            DataType::Int8 => serializer.collect_seq(primitive_array_iter::<Int8Type>(array)),
160            DataType::Int16 => serializer.collect_seq(primitive_array_iter::<Int16Type>(array)),
161            DataType::Int32 => serializer.collect_seq(primitive_array_iter::<Int32Type>(array)),
162            DataType::Int64 => serializer.collect_seq(primitive_array_iter::<Int64Type>(array)),
163            DataType::Date32 => serializer.collect_seq(
164                primitive_array_iter::<Date32Type>(array)
165                    .map(|opt| opt.map(|x| i64::from(x) * MILLISECONDS_IN_DAY)),
166            ),
167            DataType::Date64 => serializer.collect_seq(primitive_array_iter::<Date64Type>(array)),
168            DataType::Timestamp(TimeUnit::Second, _) => {
169                // Timestamps should be serialized to JSON as milliseconds.
170                serializer.collect_seq(
171                    primitive_array_iter::<TimestampSecondType>(array)
172                        .map(|opt| opt.map(|x| x * 1_000)),
173                )
174            }
175            DataType::Timestamp(TimeUnit::Millisecond, _) => {
176                // Timestamps should be serialized to JSON as milliseconds.
177                serializer.collect_seq(primitive_array_iter::<TimestampMillisecondType>(array))
178            }
179            DataType::Timestamp(TimeUnit::Microsecond, _) => {
180                // Timestamps should be serialized to JSON as milliseconds.
181                serializer.collect_seq(
182                    primitive_array_iter::<TimestampMicrosecondType>(array)
183                        .map(|opt| opt.map(|x| x / 1_000)),
184                )
185            }
186            DataType::Timestamp(TimeUnit::Nanosecond, _) => {
187                // Timestamps should be serialized to JSON as milliseconds.
188                serializer.collect_seq(
189                    primitive_array_iter::<TimestampNanosecondType>(array)
190                        .map(|opt| opt.map(|x| x / 1_000_000)),
191                )
192            }
193            DataType::UInt8 => serializer.collect_seq(primitive_array_iter::<UInt8Type>(array)),
194            DataType::UInt16 => serializer.collect_seq(primitive_array_iter::<UInt16Type>(array)),
195            DataType::UInt32 => serializer.collect_seq(primitive_array_iter::<UInt32Type>(array)),
196            DataType::UInt64 => serializer.collect_seq(primitive_array_iter::<UInt64Type>(array)),
197            DataType::Float32 => {
198                serialize_floats_and_collect_entities::<S, Float32Type>(serializer, array, self.1)
199            }
200            DataType::Float64 => {
201                serialize_floats_and_collect_entities::<S, Float64Type>(serializer, array, self.1)
202            }
203            _ => Err(S::Error::custom("unsupported arrow datatype")),
204        }
205    }
206}

208fn primitive_array_iter<T>(array: &dyn Array) -> impl Iterator<Item = Option<T::Native>> + '_
209where
210    T: ArrowPrimitiveType,
211    <T as ArrowPrimitiveType>::Native: Clone,
212{
213    array
214        .as_any()
215        .downcast_ref::<PrimitiveArray<T>>()
216        .unwrap()
217        .iter()
218}

220fn serialize_floats_and_collect_entities<S, T>(
221    serializer: S,
222    array: &dyn Array,
223    entities_ref: &RefCell<Option<Entities>>,
224) -> Result<S::Ok, S::Error>
225where
226    S: Serializer,
227    T: ArrowPrimitiveType,
228    <T as ArrowPrimitiveType>::Native: Float + Serialize,
229{
230    let array = array.as_any().downcast_ref::<PrimitiveArray<T>>().unwrap();
231    let mut seq = serializer.serialize_seq(Some(array.len()))?;
232    let mut entities = Entities::default();
233    for (i, el) in array.iter().enumerate() {
234        seq.serialize_element(&el)?;
235        match el {
236            Some(x) if x.is_nan() => entities.nan.push(i),
237            Some(x) if x.is_infinite() && x.is_sign_positive() => entities.inf.push(i),
238            Some(x) if x.is_infinite() && x.is_sign_negative() => entities.neg_inf.push(i),
239            _ => {}
240        }
241    }
242    if !entities.nan.is_empty() || !entities.inf.is_empty() || !entities.neg_inf.is_empty() {
243        *entities_ref.borrow_mut() = Some(entities);
244    }
245    seq.end()
246}

277    #[test]
278    fn round_trip_small() {
279        let f = Frame {
280            name: "many_types".to_string(),
281            ref_id: Some("A".to_string()),
282            meta: Some(Metadata {
283                custom: json!({"Hi": "there"}).as_object_mut().map(std::mem::take),
284                ..Default::default()
285            }),
286            fields: vec![
287                vec![-128i8, -128, 0, 127, 127].into_field("int8_values"),
288                vec!["foo", "bar", "baz", "qux", "quux"].into_field("string_values"),
289                vec![10000i32, 20000, 30000, 40000, 50000].into_field("int32_values"),
290                vec![10000u32, 20000, 30000, 40000, 50000].into_field("uint32_values"),
291                vec![1.0f32, 2.0, f32::NAN, f32::INFINITY, f32::NEG_INFINITY]
292                    .into_field("float32_values"),
293                vec![1.0f64, 2.0, f64::NAN, f64::INFINITY, f64::NEG_INFINITY]
294                    .into_field("float64_values"),
295                vec![Utc
296                    .with_ymd_and_hms(2024, 1, 1, 12, 13, 14)
297                    .single()
298                    .unwrap()]
299                .into_field("datetime_values"),
300                vec![NaiveDate::from_ymd_opt(2024, 1, 1).unwrap()].into_field("date_values"),
301            ],
302        };
303        let jdoc = to_string_pretty(&f).unwrap();
304        let parsed: Frame = from_str(&jdoc).unwrap();
305        let jdoc_again = to_string_pretty(&parsed).unwrap();
306        // Compare the JSON reprs; the internal Arrow datatypes will
307        // be different because the JSON representation is lossy
308        // (we lose timestamp representations and timezones).
309        assert_eq!(jdoc, jdoc_again);
310    }
