26impl Serialize for Frame {
27    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
28    where
29        S: Serializer,
30    {
31        let schema_fields: Vec<_> = self
32            .fields
33            .iter()
34            .map(|f| {
35                Ok(SerializableField {
36                    name: f.name.as_str(),
37                    labels: &f.labels,
38                    config: f.config.as_ref(),
39                    type_: f.type_info.simple_type(),
40                    type_info: &f.type_info,
41                })
42            })
43            .collect::<Result<_, _>>()?;
44        let ser = SerializableFrame {
45            schema: Some(SerializableFrameSchema {
46                name: &self.name,
47                ref_id: self.ref_id.as_deref(),
48                meta: &self.meta,
49                fields: &schema_fields,
50            }),
51            data: Some(SerializableFrameData {
52                fields: &self.fields,
53            }),
54        };
55        ser.serialize(serializer)
56    }
57}

93impl<'a> Serialize for SerializableFrameData<'a> {
94    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
95    where
96        S: Serializer,
97    {
98        let n_fields = self.fields.len();
99
100        let entities: Vec<_> = std::iter::repeat(RefCell::new(None))
101            .take(n_fields)
102            .collect();
103
104        let values = SerializableFrameDataValues {
105            fields: self.fields,
106            entities: &entities,
107        };
108
109        let mut map = serializer.serialize_map(None)?;
110        map.serialize_key("values")?;
111        map.serialize_value(&values)?;
112
113        if entities.iter().any(|r| r.borrow().is_some()) {
114            map.serialize_key("entities")?;
115            map.serialize_value(&entities)?;
116        }
117
118        map.end()
119    }
120}

122struct SerializableFrameDataValues<'a, 'b> {
123    fields: &'a [Field],
124    entities: &'b [RefCell<Option<Entities>>],
125}
126
127impl<'a, 'b> Serialize for SerializableFrameDataValues<'a, 'b> {
128    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
129    where
130        S: Serializer,
131    {
132        let mut seq = serializer.serialize_seq(Some(self.fields.len()))?;
133        for (arr, e) in self.fields.iter().zip(self.entities.iter()) {
134            seq.serialize_element(&SerializableArray(&*arr.values, e))?;
135        }
136        seq.end()
137    }
138}

140struct SerializableArray<'a>(&'a dyn Array, &'a RefCell<Option<Entities>>);
141
142impl<'a> Serialize for SerializableArray<'a> {
143    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
144    where
145        S: Serializer,
146    {
147        let array = self.0;
148        let len = array.len();
149        match array.data_type() {
150            DataType::Null => {
151                serializer.collect_seq(std::iter::repeat::<Option<()>>(None).take(len))
152            }
153            DataType::Boolean => serializer.collect_seq(
154                array
155                    .as_any()
156                    .downcast_ref::<BooleanArray>()
157                    .unwrap()
158                    .iter(),
159            ),
160            DataType::Utf8 | DataType::LargeUtf8 => {
161                serializer.collect_seq(array.as_any().downcast_ref::<StringArray>().unwrap().iter())
162            }
163            DataType::Int8 => serializer.collect_seq(primitive_array_iter::<Int8Type>(array)),
164            DataType::Int16 => serializer.collect_seq(primitive_array_iter::<Int16Type>(array)),
165            DataType::Int32 => serializer.collect_seq(primitive_array_iter::<Int32Type>(array)),
166            DataType::Int64 => serializer.collect_seq(primitive_array_iter::<Int64Type>(array)),
167            DataType::Date32 => serializer.collect_seq(
168                primitive_array_iter::<Date32Type>(array)
169                    .map(|opt| opt.map(|x| i64::from(x) * MILLISECONDS_IN_DAY)),
170            ),
171            DataType::Date64 => serializer.collect_seq(primitive_array_iter::<Date64Type>(array)),
172            DataType::Timestamp(TimeUnit::Second, _) => {
173                // Timestamps should be serialized to JSON as milliseconds.
174                serializer.collect_seq(
175                    primitive_array_iter::<TimestampSecondType>(array)
176                        .map(|opt| opt.map(|x| x * 1_000)),
177                )
178            }
179            DataType::Timestamp(TimeUnit::Millisecond, _) => {
180                // Timestamps should be serialized to JSON as milliseconds.
181                serializer.collect_seq(primitive_array_iter::<TimestampMillisecondType>(array))
182            }
183            DataType::Timestamp(TimeUnit::Microsecond, _) => {
184                // Timestamps should be serialized to JSON as milliseconds.
185                serializer.collect_seq(
186                    primitive_array_iter::<TimestampMicrosecondType>(array)
187                        .map(|opt| opt.map(|x| x / 1_000)),
188                )
189            }
190            DataType::Timestamp(TimeUnit::Nanosecond, _) => {
191                // Timestamps should be serialized to JSON as milliseconds.
192                serializer.collect_seq(
193                    primitive_array_iter::<TimestampNanosecondType>(array)
194                        .map(|opt| opt.map(|x| x / 1_000_000)),
195                )
196            }
197            DataType::UInt8 => serializer.collect_seq(primitive_array_iter::<UInt8Type>(array)),
198            DataType::UInt16 => serializer.collect_seq(primitive_array_iter::<UInt16Type>(array)),
199            DataType::UInt32 => serializer.collect_seq(primitive_array_iter::<UInt32Type>(array)),
200            DataType::UInt64 => serializer.collect_seq(primitive_array_iter::<UInt64Type>(array)),
201            DataType::Float32 => {
202                serialize_floats_and_collect_entities::<S, Float32Type>(serializer, array, self.1)
203            }
204            DataType::Float64 => {
205                serialize_floats_and_collect_entities::<S, Float64Type>(serializer, array, self.1)
206            }
207            _ => Err(S::Error::custom("unsupported arrow datatype")),
208        }
209    }
210}

212fn primitive_array_iter<T>(array: &dyn Array) -> impl Iterator<Item = Option<T::Native>> + '_
213where
214    T: ArrowPrimitiveType,
215    <T as ArrowPrimitiveType>::Native: Clone,
216{
217    array
218        .as_any()
219        .downcast_ref::<PrimitiveArray<T>>()
220        .unwrap()
221        .iter()
222}
223
224fn serialize_floats_and_collect_entities<S, T>(
225    serializer: S,
226    array: &dyn Array,
227    entities_ref: &RefCell<Option<Entities>>,
228) -> Result<S::Ok, S::Error>
229where
230    S: Serializer,
231    T: ArrowPrimitiveType,
232    <T as ArrowPrimitiveType>::Native: Float + Serialize,
233{
234    let array = array.as_any().downcast_ref::<PrimitiveArray<T>>().unwrap();
235    let mut seq = serializer.serialize_seq(Some(array.len()))?;
236    let mut entities = Entities::default();
237    for (i, el) in array.iter().enumerate() {
238        seq.serialize_element(&el)?;
239        match el {
240            Some(x) if x.is_nan() => entities.nan.push(i),
241            Some(x) if x.is_infinite() && x.is_sign_positive() => entities.inf.push(i),
242            Some(x) if x.is_infinite() && x.is_sign_negative() => entities.neg_inf.push(i),
243            _ => {}
244        }
245    }
246    if !entities.nan.is_empty() || !entities.inf.is_empty() || !entities.neg_inf.is_empty() {
247        *entities_ref.borrow_mut() = Some(entities);
248    }
249    seq.end()
250}

252#[derive(Clone, Debug, Default, Deserialize, Serialize)]
253pub(crate) struct Entities {
254    #[serde(default, rename = "NaN", skip_serializing_if = "Vec::is_empty")]

256    #[serde(default, rename = "Inf", skip_serializing_if = "Vec::is_empty")]
257    pub(crate) inf: Vec<usize>,
258    #[serde(default, rename = "NegInf", skip_serializing_if = "Vec::is_empty")]
259    pub(crate) neg_inf: Vec<usize>,
260}

282    fn round_trip_small() {
283        let f = Frame {
284            name: "many_types".to_string(),
285            ref_id: Some("A".to_string()),
286            meta: Some(Metadata {
287                custom: json!({"Hi": "there"}).as_object_mut().map(std::mem::take),
288                ..Default::default()
289            }),
290            fields: vec![
291                vec![-128i8, -128, 0, 127, 127].into_field("int8_values"),
292                vec!["foo", "bar", "baz", "qux", "quux"].into_field("string_values"),
293                vec![10000i32, 20000, 30000, 40000, 50000].into_field("int32_values"),
294                vec![10000u32, 20000, 30000, 40000, 50000].into_field("uint32_values"),
295                vec![1.0f32, 2.0, f32::NAN, f32::INFINITY, f32::NEG_INFINITY]
296                    .into_field("float32_values"),
297                vec![1.0f64, 2.0, f64::NAN, f64::INFINITY, f64::NEG_INFINITY]
298                    .into_field("float64_values"),
299                vec![Utc
300                    .with_ymd_and_hms(2024, 1, 1, 12, 13, 14)
301                    .single()
302                    .unwrap()]
303                .into_field("datetime_values"),
304                vec![NaiveDate::from_ymd_opt(2024, 1, 1).unwrap()].into_field("date_values"),
305            ],
306        };
307        let jdoc = to_string_pretty(&f).unwrap();
308        let parsed: Frame = from_str(&jdoc).unwrap();
309        let jdoc_again = to_string_pretty(&parsed).unwrap();
310        // Compare the JSON reprs; the internal Arrow datatypes will
311        // be different because the JSON representation is lossy
312        // (we lose timestamp representations and timezones).
313        assert_eq!(jdoc, jdoc_again);
314    }
