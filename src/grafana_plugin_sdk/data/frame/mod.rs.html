27/// A structured, two-dimensional data frame.
28///
29/// `Frame`s can be created manually using [`Frame::new`] if desired.
30/// Alternatively, the [`IntoFrame`] trait (and its reciprocal, [`FromFields`])
31/// provide a convenient way to create a frame from an iterator of [`Field`]s.
32///
33/// Frames generally can't be passed back to the SDK without first being checked
34/// that they are valid. The [`Frame::check`] method performs all required checks
35/// (e.g. that all fields have the same length), and returns a [`CheckedFrame<'_>`]
36/// which contains a reference to the original frame. This can then be used
37/// throughout the rest of the SDK (e.g. to be sent back to Grafana).
38///
39/// # Examples
40///
41/// Creating a Frame using the [`Frame::new`]:
42///
43/// ```rust
44/// use grafana_plugin_sdk::{
45///     data::{Field, Frame},
46///     prelude::*,
47/// };
48///
49/// let field = [1_u32, 2, 3].into_field("x");
50/// let frame = Frame::new("new")
51///     .with_field(field);
52/// ```
53///
54/// Using the [`IntoFrame`] trait:
55///
56/// ```rust
57/// use grafana_plugin_sdk::prelude::*;
58///
59/// let frame = [
60///     [1_u32, 2, 3].into_field("x"),
61///     ["a", "b", "c"].into_field("y"),
62/// ]
63/// .into_frame("super convenient");
64/// ```
65///
66/// Fields can be accessed using either [`Frame::fields`] and [`Frame::fields_mut`], or
67/// by using the [`Index`][std::ops::Index] and [`IndexMut`][std::ops::IndexMut] implementations
68/// with field indexes or names:
69///
70/// ```rust
71/// use grafana_plugin_sdk::prelude::*;
72///
73/// let frame = [
74///     [1_u32, 2, 3].into_field("x"),
75///     ["a", "b", "c"].into_field("y"),
76/// ]
77/// .into_frame("frame");
78///
79/// assert_eq!(
80///     frame.fields()[0].name,
81///     "x",
82/// );
83/// assert_eq!(
84///     frame.fields()[1],
85///     frame["y"],
86/// );
87/// ```
88#[derive(Debug)]
89#[cfg_attr(test, derive(PartialEq))]
90pub struct Frame {
91    /// The name of this frame.
92    pub name: String,
93
94    /// Optional metadata describing this frame.
95    ///
96    /// This can include custom metadata.
97    pub meta: Option<Metadata>,
98
99    /// The ID of this frame, used by the Grafana frontend.
100    ///
101    /// This does not usually need to be provided; it will be filled
102    /// in by the SDK when the Frame is returned to Grafana.
103    pub(crate) ref_id: Option<String>,
104
105    /// The fields of this frame.
106    ///
107    /// The data in all fields must be of the same length, but may have different types.
108    fields: Vec<Field>,
109}

225    /// Check this unchecked Frame, returning a [`CheckedFrame`] ready for serializing.
226    ///
227    /// `Frame`s may be in an intermediate state whilst being constructed (for example,
228    /// their field lengths may differ). Calling this method validates the frame and
229    /// returns a `CheckedFrame` if successful. Checked frames can then be used
230    /// throughout the rest of the SDK (e.g. to be sent back to Grafana).
231    ///
232    /// # Errors
233    ///
234    /// Returns an error if the fields of `self` do not all have the same length.
235    ///
236    /// # Example
237    ///
238    /// ```rust
239    /// use grafana_plugin_sdk::prelude::*;
240    ///
241    /// assert!(
242    ///     [
243    ///         [1_u32, 2, 3].into_field("x"),
244    ///         ["a", "b", "c"].into_field("y"),
245    ///     ]
246    ///     .into_frame("frame")
247    ///     .check()
248    ///     .is_ok()
249    /// );
250    ///
251    /// assert!(
252    ///     [
253    ///         [1_u32, 2, 3, 4].into_field("x"),
254    ///         ["a", "b", "c"].into_field("y"),
255    ///     ]
256    ///     .into_frame("frame")
257    ///     .check()
258    ///     .is_err()
259    /// );
260    /// ```
261    pub fn check(&self) -> Result<CheckedFrame<'_>, Error> {
262        if self.fields.iter().map(|x| x.values.len()).all_equal() {
263            Ok(CheckedFrame(self))
264        } else {
265            Err(Error::FieldLengthMismatch {
266                lengths: self
267                    .fields
268                    .iter()
269                    .map(|x| (x.name.to_string(), x.values.len()))
270                    .collect(),
271            })
272        }
273    }

395/// A reference to a checked and verified [`Frame`], which is ready for serialization.
396pub struct CheckedFrame<'a>(&'a Frame);
397
398impl CheckedFrame<'_> {
399    /// Serialize this frame to JSON, returning the serialized bytes.
400    ///
401    /// Unlike the `Serialize` impl, this method allows for customization of the fields included.
402    pub(crate) fn to_json(&self, include: FrameInclude) -> Result<Vec<u8>, serde_json::Error> {
403        let schema_fields: Vec<_> = self
404            .0
405            .fields
406            .iter()
407            .map(|f| SerializableField {
408                name: &f.name,
409                labels: &f.labels,
410                config: f.config.as_ref(),
411                type_: f.type_info.simple_type(),
412                type_info: &f.type_info,
413            })
414            .collect();
415        let ser = SerializableFrame {
416            schema: matches!(include, FrameInclude::All | FrameInclude::SchemaOnly).then(|| {
417                SerializableFrameSchema {
418                    name: &self.0.name,
419                    ref_id: self.0.ref_id.as_deref(),
420                    meta: &self.0.meta,
421                    fields: &schema_fields,
422                }
423            }),
424            data: matches!(include, FrameInclude::All | FrameInclude::DataOnly).then(|| {
425                SerializableFrameData {
426                    fields: &self.0.fields,
427                }
428            }),
429        };
430        serde_json::to_vec(&ser)
431    }
432}

434/// Convenience trait for converting iterators of [`Field`]s into a [`Frame`].
435#[cfg_attr(docsrs, doc(notable_trait))]
436pub trait IntoFrame {
437    /// Create a [`Frame`] with the given name from `self`.
438    fn into_frame(self, name: impl Into<String>) -> Frame;
439}
440
441impl<T> IntoFrame for T
442where
443    T: IntoIterator<Item = Field>,
444{
445    fn into_frame(self, name: impl Into<String>) -> Frame {
446        Frame {
447            name: name.into(),
448            fields: self.into_iter().collect(),
449            meta: None,
450            ref_id: None,
451        }
452    }
453}

455/// Convenience trait for creating a [`Frame`] from an iterator of [`Field`]s.
456///
457/// This is the inverse of [`IntoFrame`] and is defined for all implementors of that trait.
458#[cfg_attr(docsrs, doc(notable_trait))]
459pub trait FromFields<T: IntoFrame> {
460    /// Create a [`Frame`] with the given name from `fields`.
461    fn from_fields(name: impl Into<String>, fields: T) -> Frame;
462}
463
464impl<T: IntoFrame> FromFields<T> for Frame {
465    fn from_fields(name: impl Into<String>, fields: T) -> Frame {
466        fields.into_frame(name)
467    }
468}
