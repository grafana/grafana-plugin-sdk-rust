27// Deserialization from the JSON representation for [`Frame`]s.
28//
29// This follows the [Manually deserializing a struct](https://serde.rs/deserialize-struct.html)
30// example in the [`serde`] docs, and uses [`RawValue`] to avoid intermediate buffering
31// of arrays during deserialization.
32impl<'de> Deserialize<'de> for Frame {
33    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
34    where
35        D: Deserializer<'de>,
36    {
37        #[derive(Deserialize)]
38        #[serde(field_identifier, rename_all = "lowercase")]
39        enum Field {
40            Schema,
41            Data,
42        }
43
44        struct FrameVisitor;
45
46        impl<'de> Visitor<'de> for FrameVisitor {
47            type Value = Frame;
48
49            fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
50                formatter.write_str("stuct Frame")
51            }
52
53            fn visit_map<V>(self, mut map: V) -> Result<Self::Value, V::Error>
54            where
55                V: MapAccess<'de>,
56            {
57                let mut schema: Option<Schema> = None;
58                let mut raw_data: Option<RawData> = None;
59
60                while let Some(key) = map.next_key()? {
61                    match key {
62                        Field::Schema => {
63                            if schema.is_some() {
64                                return Err(Error::duplicate_field("schema"));
65                            }
66                            schema = Some(map.next_value()?);
67                        }
68                        Field::Data => {
69                            if raw_data.is_some() {
70                                return Err(Error::duplicate_field("data"));
71                            }
72                            raw_data = Some(map.next_value()?);
73                        }
74                    }
75                }
76                let schema = schema.ok_or_else(|| Error::missing_field("schema"))?;
77                let raw_data = raw_data.ok_or_else(|| Error::missing_field("data"))?;
78                let data: Data = (&schema, raw_data)
79                    .try_into()
80                    .map_err(|e| Error::custom(format!("invalid values: {}", e)))?;
81
82                Ok(Frame {
83                    name: schema.name,
84                    ref_id: Some(schema.ref_id),
85                    meta: schema.meta,
86                    fields: schema
87                        .fields
88                        .into_iter()
89                        .zip(data.values)
90                        .map(|(f, values)| crate::data::field::Field {
91                            name: f.name,
92                            labels: f.labels,
93                            config: f.config,
94                            values,
95                            type_info: f.type_info,
96                        })
97                        .collect(),
98                })
99            }
100        }
101
102        const FIELDS: &[&str] = &["schema", "data"];
103        deserializer.deserialize_struct("Frame", FIELDS, FrameVisitor)
104    }
105}

132#[derive(Debug, Deserialize)]
133struct RawData<'a> {
134    #[serde(borrow, default)]
135    values: Vec<&'a serde_json::value::RawValue>,
136    #[serde(default)]
137    entities: Option<Vec<Option<Entities>>>,
138}

140#[derive(Debug)]
141struct Data {
142    values: Vec<Arc<dyn Array>>,
143}
144
145impl TryFrom<(&'_ Schema, RawData<'_>)> for Data {
146    type Error = serde_json::Error;
147
148    /// Create `Data` from the schema and data objects found in the JSON representation.
149    ///
150    /// This handles deserializing each of the arrays in `values` with the correct datatypes,
151    /// replacing any 'entities' (`NaN`, `Inf`, `-Inf`) in those arrays, and converting any
152    /// which require it (e.g. timestamps).
153    fn try_from((schema, data): (&'_ Schema, RawData<'_>)) -> Result<Self, Self::Error> {
154        // Handle entity replacement, return values.
155        let fields = schema.fields.iter();
156        let values = data.values.into_iter();
157        let entities = data
158            .entities
159            .unwrap_or_else(|| vec![None; fields.len()])
160            .into_iter();
161        Ok(Self {
162            values: fields
163                .zip(values)
164                .zip(entities)
165                .map(|((f, v), e)| {
166                    let s = v.get();
167                    let arr: Arc<dyn Array> = match f.type_info.frame {
168                        TypeInfoType::Int8 => parse_primitive_array::<Int8Array, Int8Type>(s)?,
169                        TypeInfoType::Int16 => parse_primitive_array::<Int16Array, Int16Type>(s)?,
170                        TypeInfoType::Int32 => parse_primitive_array::<Int32Array, Int32Type>(s)?,
171                        TypeInfoType::Int64 => parse_primitive_array::<Int64Array, Int64Type>(s)?,
172                        TypeInfoType::UInt8 => parse_primitive_array::<UInt8Array, UInt8Type>(s)?,
173                        TypeInfoType::UInt16 => {
174                            parse_primitive_array::<UInt16Array, UInt16Type>(s)?
175                        }
176                        TypeInfoType::UInt32 => {
177                            parse_primitive_array::<UInt32Array, UInt32Type>(s)?
178                        }
179                        TypeInfoType::UInt64 => {
180                            parse_primitive_array::<UInt64Array, UInt64Type>(s)?
181                        }
182                        // TypeInfoType::Float16 => {
183                        //     parse_array_with_entities::<Float16Array, Float16Type>(s, e)?
184                        // }
185                        TypeInfoType::Float32 => {
186                            parse_array_with_entities::<Float32Array, Float32Type>(s, e)?
187                        }
188                        TypeInfoType::Float64 => {
189                            parse_array_with_entities::<Float64Array, Float64Type>(s, e)?
190                        }
191                        TypeInfoType::String => parse_string_array(s)?,
192                        TypeInfoType::Bool => parse_bool_array(s)?,
193                        TypeInfoType::Time => parse_timestamp_array(s)?,
194                    };
195                    Ok(arr)
196                })
197                .collect::<Result<Vec<_>, _>>()?,
198        })
199    }
200}

202/// Parse a JSON array containing elements of `U` into a mutable Arrow array `T`.
203///
204/// # Errors
205///
206/// Returns an error if the string is invalid JSON, if the elements of
207/// the array are not of type `U`,  or if the Arrow buffer could not be
208/// created.
209fn parse_primitive_array<T, U>(s: &str) -> Result<Arc<dyn Array>, serde_json::Error>
210where
211    T: From<Vec<Option<U::Native>>> + Array + 'static,
212    U: ArrowPrimitiveType,
213    U::Native: DeserializeOwned,
214{
215    let v: Vec<Option<U::Native>> = serde_json::from_str(s)?;
216    Ok(Arc::new(T::from(v)))
217}
218
219fn parse_timestamp_array(s: &str) -> Result<Arc<dyn Array>, serde_json::Error> {
220    let v: Vec<Option<i64>> = serde_json::from_str::<Vec<Option<i64>>>(s)?
221        .into_iter()
222        .map(|opt| opt.map(|x| x * 1_000_000))
223        .collect();
224    Ok(Arc::new(TimestampNanosecondArray::from(v)))
225}
226
227fn parse_string_array(s: &str) -> Result<Arc<dyn Array>, serde_json::Error> {
228    let v: Vec<Option<String>> = serde_json::from_str(s)?;
229    Ok(Arc::new(StringArray::from(v)))
230}
231
232fn parse_bool_array(s: &str) -> Result<Arc<dyn Array>, serde_json::Error> {
233    let v: Vec<Option<bool>> = serde_json::from_str(s)?;
234    Ok(Arc::new(BooleanArray::from(v)))
235}

237/// Parse a JSON array containing elements of `U` into a mutable Arrow array `T`,
238/// then substitutes any sentinel `entities`.
239///
240/// # Errors
241///
242/// Returns an error if the string is invalid JSON, if the elements of
243/// the array are not of type `U`,  or if the Arrow buffer could not be
244/// created.
245///
246/// # Panics
247///
248/// Panics if any of the indexes in `entities` are greater than the length
249/// of the parsed array.
250fn parse_array_with_entities<T, U>(
251    s: &str,
252    entities: Option<Entities>,
253) -> Result<Arc<dyn Array>, serde_json::Error>
254where
255    T: From<Vec<Option<U::Native>>> + Array + 'static,
256    U: ArrowPrimitiveType,
257    U::Native: DeserializeOwned + Float,
258{
259    let de = DeArray::<T, U>::new(entities);
260    let mut deserializer = serde_json::Deserializer::from_str(s);
261    Ok(Arc::new(de.deserialize(&mut deserializer)?))
262}

264/// Helper struct used to deserialize a sequence into an Arrow `Array`.
265#[derive(Debug)]
266struct DeArray<T, U> {
267    entities: Option<Entities>,
268    // The type of the final array.
269    t: PhantomData<T>,
270    // The type of the elements in the array.
271    u: PhantomData<U>,
272}
273
274impl<T, U> DeArray<T, U> {
275    fn new(entities: Option<Entities>) -> Self {
276        Self {
277            entities,
278            t: PhantomData,
279            u: PhantomData,
280        }
281    }
282}

284// Deserialization for mutable Arrow arrays.
285//
286// Constructs a mutable array from a sequence of valid values.
287// All of the `Mutable` variants of Arrow arrays implement `TryPush<Option<U>>`
288// for some relevant `U`, and here we just impose that the `U` is `Deserialize`
289// and gradually build up the array.
290impl<'de, T, U> DeserializeSeed<'de> for DeArray<T, U>
291where
292    T: From<Vec<Option<U::Native>>> + Array + 'static,
293    U: ArrowPrimitiveType,
294    U::Native: Deserialize<'de> + Float,
295{
296    type Value = T;
297
298    fn deserialize<D>(self, deserializer: D) -> Result<T, D::Error>
299    where
300        D: Deserializer<'de>,
301    {
302        struct ArrayVisitor<T, U>(Option<Entities>, PhantomData<T>, PhantomData<U>);
303
304        impl<'de, T, U> Visitor<'de> for ArrayVisitor<T, U>
305        where
306            T: From<Vec<Option<U::Native>>> + Array + 'static,
307            U: ArrowPrimitiveType,
308            U::Native: Deserialize<'de> + Float,
309        {
310            type Value = T;
311
312            fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
313                formatter.write_str("a heterogeneous array of compatible values")
314            }
315
316            fn visit_seq<S>(self, mut seq: S) -> Result<Self::Value, S::Error>
317            where
318                S: SeqAccess<'de>,
319            {
320                let mut array = Vec::with_capacity(seq.size_hint().unwrap_or(0));
321                while let Some(opt) = seq.next_element::<Option<U::Native>>()? {
322                    array.push(opt);
323                }
324
325                if let Some(e) = self.0 {
326                    e.nan.iter().for_each(|idx| {
327                        if let Some(v) = array.get_mut(*idx) {
328                            *v = Some(U::Native::nan());
329                        }
330                    });
331                    e.inf.iter().for_each(|idx| {
332                        if let Some(v) = array.get_mut(*idx) {
333                            *v = Some(U::Native::infinity());
334                        }
335                    });
336                    e.neg_inf.iter().for_each(|idx| {
337                        if let Some(v) = array.get_mut(*idx) {
338                            *v = Some(U::Native::neg_infinity());
339                        }
340                    });
341                }
342                Ok(array.into())
343            }
344        }
345
346        deserializer.deserialize_seq(ArrayVisitor::<T, U>(
347            self.entities,
348            PhantomData,
349            PhantomData,
350        ))
351    }
352}
