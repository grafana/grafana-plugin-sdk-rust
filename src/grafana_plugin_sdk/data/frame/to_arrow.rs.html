1//! Conversion of [`Frame`][crate::data::Frame]s to the Arrow IPC format.
2use std::{collections::HashMap, sync::Arc};
3
4use arrow::{datatypes::Schema, ipc::writer::FileWriter, record_batch::RecordBatch};
5use thiserror::Error;
6
7use crate::data::{field::Field, frame::CheckedFrame};
8
9/// Errors occurring when serializing a [`Frame`][crate::data::Frame] to the Arrow IPC format.
10#[derive(Debug, Error)]
11#[non_exhaustive]
12pub enum Error {
13    /// An error occurred converting the frame's metadata to JSON.
14    #[error("Error serializing metadata")]
15    Json(#[from] serde_json::Error),
16    /// An error occurred creating the Arrow record batch.
17    #[error("Error creating record batch: {0}")]
18    CreateRecordBatch(arrow::error::ArrowError),
19    /// An error occurred when attempting to create or write data to the output buffer.
20    #[error("Error writing data to Arrow buffer")]
21    WriteBuffer(arrow::error::ArrowError),
22}
23
24impl CheckedFrame<'_> {
25    /// Create an Arrow [`Schema`] for this Frame.
26    ///
27    /// If `ref_id` is provided, it is passed down to the various conversion
28    /// function and takes precedence over the `ref_id` set on the frame.
29    fn arrow_schema(&self, ref_id: Option<String>) -> Result<Schema, serde_json::Error> {
30        let fields: Vec<_> = self
31            .0
32            .fields
33            .iter()
34            .map(Field::to_arrow_field)
35            .collect::<Result<_, _>>()?;
36        let mut metadata: HashMap<String, String> = [
37            ("name".to_string(), self.0.name.to_string()),
38            (
39                "refId".to_string(),
40                ref_id.unwrap_or_else(|| {
41                    self.0
42                        .ref_id
43                        .as_ref()
44                        .map(|x| x.to_string())
45                        .unwrap_or_default()
46                }),
47            ),
48        ]
49        .into_iter()
50        .collect();
51        if let Some(meta) = &self.0.meta {
52            metadata.insert("meta".to_string(), serde_json::to_string(&meta)?);
53        }
54        Ok(Schema::new_with_metadata(fields, metadata))
55    }
56
57    /// Convert this [`Frame`][crate::data::Frame] to Arrow using the IPC format.
58    ///
59    /// If `ref_id` is provided, it is passed down to the various conversion
60    /// function and takes precedence over the `ref_id` set on the frame.
61    pub(crate) fn to_arrow(&self, ref_id: Option<String>) -> Result<Vec<u8>, Error> {
62        let schema = Arc::new(self.arrow_schema(ref_id)?);
63
64        let records = if self.0.fields.is_empty() {
65            None
66        } else {
67            Some(
68                RecordBatch::try_new(
69                    Arc::clone(&schema),
70                    self.0.fields.iter().map(|f| f.values.clone()).collect(),
71                )
72                .map_err(Error::CreateRecordBatch)?,
73            )
74        };
75
76        let mut buf = Vec::new();
77        {
78            let mut writer =
79                FileWriter::try_new_buffered(&mut buf, &schema).map_err(Error::WriteBuffer)?;
80            if let Some(records) = records {
81                writer.write(&records).map_err(Error::WriteBuffer)?;
82            }
83            writer.finish().map_err(Error::WriteBuffer)?;
84        }
85        Ok(buf)
86    }
87}
