1//! Error types returned by the SDK.
2use arrow::datatypes::DataType;
3use itertools::Itertools;
4use thiserror::Error;
5
6use super::frame::to_arrow;
7
8/// Errors that can occur when interacting with the Grafana plugin SDK.
9#[derive(Debug, Error)]
10#[non_exhaustive]
11pub enum Error {
12    /// An error has occurred when serializing to Arrow IPC format.
13    #[error("Arrow serialization error: {0}")]
14    ArrowSerialization(#[from] to_arrow::Error),
15
16    /// There is a datatype mismatch in a field.
17    ///
18    /// This can happen when calling [`Field::set_values`][crate::data::Field::set_values] with an array whose datatype
19    /// does not match the existing array.
20    #[error(
21        "Data type mismatch in field {} (existing: {existing:?}, new: {new:?})",
22        field
23    )]
24    DataTypeMismatch {
25        /// The existing datatype of the field.
26        existing: DataType,
27        /// The datatype of the new data.
28        new: DataType,
29        /// The name of the field.
30        field: String,
31    },
32
33    /// Occurs when a frame had mismatched field lengths while checking.
34    #[error(
35        "Frame field length mismatch: {}",
36        .lengths.iter().map(|x| format!("{} ({})", x.0, x.1)).join(", ")
37    )]
38    FieldLengthMismatch {
39        /// The names and lengths of the fields in the `Frame`.
40        lengths: Vec<(String, usize)>,
41    },
42
43    /// A field was created using an Arrow array with a datatype unsupported by Grafana.
44    #[error("Unsupported Arrow data type: {0:?}")]
45    UnsupportedArrowDataType(DataType),
46}
