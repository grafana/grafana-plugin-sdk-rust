1//! Data types used throughout the SDK.
2//!
3//! Note that many of the types in this module are marked as `#[non_exhaustive]` and cannot
4//! be constructed using struct expressions. This is because Grafana may choose to add
5//! additional fields or variants at any time. Instead, use the constructor (if available) or
6//! create a mutable default value using `Default::default()` and modify any fields.
7//!
8//! For example:
9//!
10//! ```
11//! use grafana_plugin_sdk::{
12//!     data::{Field, FieldConfig, Frame, Metadata, Notice, Severity},
13//!     prelude::*,
14//! };
15//!
16//! let mut field_config = FieldConfig::default();
17//! field_config.display_name_from_ds = Some("X".to_string());
18//! let field = ["a", "b", "c"]
19//!     .into_field("x")
20//!     .with_config(field_config);
21//!
22//! let mut notice = Notice::new("warning! this is important".to_string());
23//! notice.severity = Some(Severity::Warning);
24//! let mut metadata = Metadata::default();
25//! metadata.path = Some("my path".to_string());
26//! metadata.notices = Some(vec![notice]);
27//!
28//! let frame = Frame::new("my frame")
29//!     .with_field(field)
30//!     .with_metadata(metadata);
31//! # assert_eq!(frame.fields()[0].config.as_ref().unwrap().display_name_from_ds, Some("X".to_string()));
32//! ```

45/// A wrapper around an `Option<f64>` used in various backend data structures, with custom NaN and Infinity serialization.
46#[derive(Clone, Copy, Debug, PartialEq, Serialize, Deserialize)]
47pub struct ConfFloat64(#[serde(serialize_with = "serialize_conf_float64")] Option<f64>);
48
49fn serialize_conf_float64<S: Serializer>(val: &Option<f64>, s: S) -> Result<S::Ok, S::Error> {
50    if let Some(f) = val {
51        if f.is_nan() || f.is_infinite() {
52            s.serialize_none()
53        } else {
54            s.serialize_f64(*f)
55        }
56    } else {
57        s.serialize_none()
58    }
59}
