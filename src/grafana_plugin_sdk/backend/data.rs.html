16/// A request for data made by Grafana.
17///
18/// Details of the request source can be found in `plugin_context`,
19/// while the actual plugins themselves are in `queries`.
20#[derive(Debug)]
21#[non_exhaustive]
22pub struct InnerQueryDataRequest<Q, IS, JsonData, SecureJsonData>
23where
24    Q: DeserializeOwned,
25    JsonData: fmt::Debug + DeserializeOwned,
26    SecureJsonData: DeserializeOwned,
27    IS: InstanceSettings<JsonData, SecureJsonData>,
28{
29    /// Details of the plugin instance from which the request originated.
30    ///
31    /// If the request originates from a datasource instance, this will
32    /// include details about the datasource instance in the
33    /// `data_source_instance_settings` field.
34    pub plugin_context: backend::PluginContext<IS, JsonData, SecureJsonData>,
35    /// Headers included along with the request by Grafana.
36    pub headers: HashMap<String, String>,
37    /// The queries requested by a user or alert.
38    ///
39    /// Each [`DataQuery`] contains a unique `ref_id` field which identifies
40    /// the query to the frontend; this should be included in the corresponding
41    /// `DataResponse` for each query.
42    pub queries: Vec<DataQuery<Q>>,
43}
44
45impl<Q, IS, JsonData, SecureJsonData> TryFrom<pluginv2::QueryDataRequest>
46    for InnerQueryDataRequest<Q, IS, JsonData, SecureJsonData>
47where
48    Q: DeserializeOwned,
49    JsonData: fmt::Debug + DeserializeOwned,
50    SecureJsonData: DeserializeOwned,
51    IS: InstanceSettings<JsonData, SecureJsonData>,
52{
53    type Error = ConvertFromError;
54    fn try_from(other: pluginv2::QueryDataRequest) -> Result<Self, Self::Error> {
55        Ok(Self {
56            plugin_context: other
57                .plugin_context
58                .ok_or(ConvertFromError::MissingPluginContext)
59                .and_then(TryInto::try_into)?,
60            headers: other.headers,
61            queries: other
62                .queries
63                .into_iter()
64                .map(DataQuery::try_from)
65                .collect::<Result<Vec<_>, _>>()?,
66        })
67    }
68}

70/// A request for data made by Grafana.
71///
72/// Details of the request source can be found in `plugin_context`,
73/// while the actual plugins themselves are in `queries`.
74///
75/// This is a convenience type alias to hide some of the complexity of
76/// the various generics involved.
77///
78/// The type parameter `Q` is the type of the query: generally you can use
79/// `Self::Query` here.
80///
81/// The type parameter `T` is the type of the plugin implementation itself,
82/// which must implement [`GrafanaPlugin`].
83pub type QueryDataRequest<Q, T> = InnerQueryDataRequest<
84    Q,
85    <<T as GrafanaPlugin>::PluginType as PluginType<
86        <T as GrafanaPlugin>::JsonData,
87        <T as GrafanaPlugin>::SecureJsonData,
88    >>::InstanceSettings,
89    <T as GrafanaPlugin>::JsonData,
90    <T as GrafanaPlugin>::SecureJsonData,
91>;

93/// A query made by Grafana to the plugin as part of a [`QueryDataRequest`].
94///
95/// The `query` field contains any fields set by the plugin's UI.
96#[derive(Debug)]
97#[non_exhaustive]
98pub struct DataQuery<Q>
99where
100    Q: DeserializeOwned,
101{
102    /// The unique identifier of the query, set by the frontend call.
103    ///
104    /// This should be included in the corresponding [`DataResponse`].
105    pub ref_id: String,
106
107    /// An identifier for the type of query.
108    ///
109    /// This can be used to distinguish different types of queries.
110    pub query_type: String,
111
112    /// The maximum number of datapoints that should be returned from a time series query.
113    pub max_data_points: i64,
114
115    /// The suggested duration between time points in a time series query.
116    pub interval: Duration,
117
118    /// The start and end of the query requested by the frontend.
119    pub time_range: TimeRange,
120
121    /// The inner query.
122    ///
123    /// This contains all of the other properties, as well as custom properties.
124    pub query: Q,
125}
126
127impl<Q> TryFrom<pluginv2::DataQuery> for DataQuery<Q>
128where
129    Q: DeserializeOwned,
130{
131    type Error = ConvertFromError;
132    fn try_from(other: pluginv2::DataQuery) -> Result<Self, Self::Error> {
133        Ok(Self {
134            ref_id: other.ref_id,
135            query_type: other.query_type,
136            max_data_points: other.max_data_points,
137            interval: Duration::from_millis(other.interval_ms as u64),
138            time_range: other
139                .time_range
140                .map(TimeRange::from)
141                .ok_or(ConvertFromError::MissingTimeRange)?,
142            query: backend::read_json_query(&other.json)?,
143        })
144    }
145}

147/// The results from a [`DataQuery`].
148#[derive(Debug)]
149pub struct DataResponse {
150    /// The unique identifier of the query, set by the frontend call.
151    ///
152    /// This is used to align queries in the request to data in the response,
153    /// and can be obtained from the [`DataQuery`].
154    ref_id: String,
155
156    /// The data returned from the query.
157    frames: Result<Vec<Vec<u8>>, data::Error>,
158}
159
160impl DataResponse {
161    /// Create a new [`DataResponse`] with the given `ref_id` and `frames`.
162    #[must_use]
163    pub fn new(ref_id: String, frames: Vec<data::CheckedFrame<'_>>) -> Self {
164        Self {
165            ref_id: ref_id.clone(),
166            frames: to_arrow(frames, &Some(ref_id)),
167        }
168    }
169}

171/// Error supertrait used in [`DataService::query_data`].
172pub trait DataQueryError: std::error::Error {
173    /// Return the `ref_id` of the incoming query to which this error corresponds.
174    ///
175    /// This allows the SDK to align queries up with any failed requests.
176    fn ref_id(self) -> String;
177
178    /// A suitable [`DataQueryStatus`] to represent this error.
179    ///
180    /// This may be used by clients to decide how this error should
181    /// be handled. For example, whether the request should be retried,
182    /// treated as a client or server error, etc.
183    ///
184    /// Defaults to [`DataQueryStatus::Unknown`] if not overridden.
185    fn status(&self) -> DataQueryStatus {
186        DataQueryStatus::Unknown
187    }
188
189    /// The source of the error.
190    ///
191    /// Defaults to [`ErrorSource::Plugin`] if not overridden.
192    fn source(&self) -> ErrorSource {
193        ErrorSource::default()
194    }
195}

197/// Status codes for [`DataQueryError`].
198///
199/// These generally correspond to HTTP status codes, but are not
200/// a 1:1 mapping: several variants may map to a single HTTP
201/// status code, and not all HTTP status codes are given.
202///
203/// To return a custom HTTP status code in more advanced scenarios,
204/// use [`DataQueryStatus::Custom`] and nest the required
205/// [`http::StatusCode`] inside.
206pub enum DataQueryStatus {
207    /// An error that should be updated to contain an accurate status code,
208    /// as none has been provided.
209    ///
210    /// HTTP status code 500.
211    Unknown,
212
213    /// The query was successful.
214    ///
215    /// HTTP status code 200.
216    OK,
217
218    /// The datasource does not recognize the client's authentication,
219    /// either because it has not been provided
220    /// or is invalid for the operation.
221    ///
222    /// HTTP status code 401.
223    Unauthorized,
224
225    /// The datasource refuses to perform the requested action for the authenticated user.
226    /// HTTP status code 403.
227    Forbidden,
228
229    /// The datasource does not have any corresponding document to return to the request.
230    ///
231    /// HTTP status code 404.
232    NotFound,
233
234    /// The client is rate limited by the datasource and should back-off before trying again.
235    ///
236    /// HTTP status code 429.
237    TooManyRequests,
238
239    /// The datasource was unable to parse the parameters or payload for the request.
240    ///
241    /// HTTP status code 400.
242    BadRequest,
243
244    /// The datasource was able to parse the payload for the request,
245    /// but it failed one or more validation checks.
246    ///
247    /// HTTP status code 400.
248    ValidationFailed,
249
250    /// The datasource acknowledges that there's an error, but that there is
251    /// nothing the client can do to fix it.
252    ///
253    /// HTTP status code 500.
254    Internal,
255
256    /// The datasource does not support the requested action.
257    /// Typically used during development of new features.
258    ///
259    /// HTTP status code 501.
260    NotImplemented,
261
262    /// The datasource did not complete the request within the required time and aborted the action.
263    ///
264    /// HTTP status code 504.
265    Timeout,
266
267    /// The datasource, while acting as a gateway or proxy, received an invalid response
268    /// from the upstream server.
269    ///
270    /// HTTP status code 502.
271    BadGateway,
272
273    /// The datasource encountered another error, best represented by
274    /// the inner status code.
275    Custom(http::StatusCode),
276}
277
278impl DataQueryStatus {
279    fn as_http(&self) -> http::StatusCode {
280        use http::StatusCode;
281        match self {
282            DataQueryStatus::Unknown => StatusCode::INTERNAL_SERVER_ERROR,
283            DataQueryStatus::OK => StatusCode::OK,
284            DataQueryStatus::Unauthorized => StatusCode::UNAUTHORIZED,
285            DataQueryStatus::Forbidden => StatusCode::FORBIDDEN,
286            DataQueryStatus::NotFound => StatusCode::NOT_FOUND,
287            DataQueryStatus::TooManyRequests => StatusCode::TOO_MANY_REQUESTS,
288            DataQueryStatus::BadRequest => StatusCode::BAD_REQUEST,
289            DataQueryStatus::ValidationFailed => StatusCode::BAD_REQUEST,
290            DataQueryStatus::Internal => StatusCode::INTERNAL_SERVER_ERROR,
291            DataQueryStatus::NotImplemented => StatusCode::NOT_IMPLEMENTED,
292            DataQueryStatus::Timeout => StatusCode::GATEWAY_TIMEOUT,
293            DataQueryStatus::BadGateway => StatusCode::BAD_GATEWAY,
294            DataQueryStatus::Custom(inner) => *inner,
295        }
296    }
297
298    fn as_i32(&self) -> i32 {
299        self.as_http().as_u16().into()
300    }
301}

303/// Used to respond for requests for data from datasources and app plugins.
304///
305/// Datasource plugins will usually want to implement this trait to perform the
306/// bulk of their processing.
307///
308/// # Example
309///
310/// ```rust
311/// use futures_util::stream::FuturesOrdered;
312/// use grafana_plugin_sdk::{backend, data, prelude::*};
313/// use thiserror::Error;
314///
315/// #[derive(Clone, Debug, GrafanaPlugin)]
316/// #[grafana_plugin(plugin_type = "datasource")]
317/// struct MyPlugin;
318///
319/// /// An error that may occur during a query.
320/// ///
321/// /// This must store the `ref_id` of the query so that Grafana can line it up.
322/// #[derive(Debug, Error)]
323/// #[error("Error querying backend for query {ref_id}: {source}")]
324/// struct QueryError {
325///     source: data::Error,
326///     ref_id: String,
327/// }
328///
329/// impl backend::DataQueryError for QueryError {
330///     fn ref_id(self) -> String {
331///         self.ref_id
332///     }
333/// }
334///
335/// #[backend::async_trait]
336/// impl backend::DataService for MyPlugin {
337///
338///     /// The type of JSON data sent from Grafana to our backend plugin.
339///     ///
340///     /// This will correspond to the `TQuery` type parameter of the frontend
341///     /// datasource.
342///     ///
343///     /// We can use `serde_json::Value` if we want to accept any JSON.
344///     type Query = serde_json::Value;
345///
346///     /// The type of error that could be returned by an individual query.
347///     type QueryError = QueryError;
348///
349///     /// The type of iterator we're returning.
350///     ///
351///     /// In general the concrete type will be impossible to name in advance,
352///     /// so the `backend::BoxDataResponseStream` type alias will be useful.
353///     type Stream = backend::BoxDataResponseStream<Self::QueryError>;
354///
355///     /// Respond to a request for data from Grafana.
356///     ///
357///     /// This request will contain zero or more queries, as well as information
358///     /// about the datasource instance on behalf of which this request is made,
359///     /// such as address, credentials, etc.
360///     ///
361///     /// Our plugin must respond to each query and return an iterator of `DataResponse`s,
362///     /// which themselves can contain zero or more `Frame`s.
363///     async fn query_data(&self, request: backend::QueryDataRequest<Self::Query, Self>) -> Self::Stream {
364///         Box::pin(
365///             request
366///                 .queries
367///                 .into_iter()
368///                 .map(|x| async {
369///                     // Here we create a single response Frame for each query.
370///                     // Frames can be created from iterators of fields using [`IntoFrame`].
371///                     Ok(backend::DataResponse::new(
372///                         x.ref_id.clone(),
373///                         // Return zero or more frames.
374///                         // A real implementation would fetch this data from a database
375///                         // or something.
376///                         vec![[
377///                             [1_u32, 2, 3].into_field("x"),
378///                             ["a", "b", "c"].into_field("y"),
379///                         ]
380///                         .into_frame("foo")
381///                         .check()
382///                         .map_err(|source| QueryError {
383///                             ref_id: x.ref_id,
384///                             source,
385///                         })?],
386///                     ))
387///                 })
388///                 .collect::<FuturesOrdered<_>>(),
389///         )
390///     }
391/// }
392/// ```
393#[tonic::async_trait]
394pub trait DataService: GrafanaPlugin {
395    /// The type of the JSON query sent from Grafana to the plugin.
396    type Query: DeserializeOwned + Send + Sync;
397
398    /// The error type that can be returned by individual queries.
399    ///
400    /// This must implement [`DataQueryError`], which allows the SDK to
401    /// align queries up with any failed requests.
402    type QueryError: DataQueryError;
403
404    /// The type of stream returned by the `query_data` method.
405    ///
406    /// This will generally be impossible to name directly, so returning the
407    /// [`BoxDataResponseStream`] type alias will probably be more convenient.
408    type Stream: Stream<Item = Result<DataResponse, Self::QueryError>> + Send;
409
410    /// Query data for an input request.
411    ///
412    /// The request will contain zero or more queries, as well as information about the
413    /// origin of the queries (such as the datasource instance) in the `plugin_context` field.
414    async fn query_data(&self, request: QueryDataRequest<Self::Query, Self>) -> Self::Stream;
415}

417/// Type alias for a boxed iterator of query responses, useful for returning from [`DataService::query_data`].
418pub type BoxDataResponseStream<E> =
419    Pin<Box<dyn Stream<Item = Result<backend::DataResponse, E>> + Send>>;
420
421/// Serialize a slice of frames to Arrow IPC format.
422///
423/// If `ref_id` is provided, it is passed down to the various conversion
424/// function and takes precedence over any `ref_id`s set on the individual frames.
425pub(crate) fn to_arrow<'a>(
426    frames: impl IntoIterator<Item = data::CheckedFrame<'a>>,
427    ref_id: &Option<String>,
428) -> Result<Vec<Vec<u8>>, data::Error> {
429    frames
430        .into_iter()
431        .map(|frame| Ok(frame.to_arrow(ref_id.clone())?))
432        .collect()
433}

435#[tonic::async_trait]
436impl<T> pluginv2::data_server::Data for T
437where
438    T: DataService + Send + Sync + 'static,
439{
440    #[tracing::instrument(skip(self), level = "debug")]
441    async fn query_data(
442        &self,
443        request: tonic::Request<pluginv2::QueryDataRequest>,
444    ) -> Result<tonic::Response<pluginv2::QueryDataResponse>, tonic::Status> {
445        let responses = DataService::query_data(
446            self,
447            request
448                .into_inner()
449                .try_into()
450                .map_err(ConvertFromError::into_tonic_status)?,
451        )
452        .await
453        .map(|resp| match resp {
454            Ok(x) => {
455                let ref_id = x.ref_id;
456                x.frames.map_or_else(
457                    |e| {
458                        (
459                            ref_id.clone(),
460                            pluginv2::DataResponse {
461                                frames: vec![],
462                                status: DataQueryStatus::Internal.as_i32(),
463                                error: e.to_string(),
464                                json_meta: vec![],
465                                error_source: ErrorSource::Plugin.to_string(),
466                            },
467                        )
468                    },
469                    |frames| {
470                        (
471                            ref_id.clone(),
472                            pluginv2::DataResponse {
473                                frames,
474                                status: DataQueryStatus::OK.as_i32(),
475                                error: "".to_string(),
476                                json_meta: vec![],
477                                error_source: "".to_string(),
478                            },
479                        )
480                    },
481                )
482            }
483            Err(e) => {
484                let status = e.status().as_i32();
485                let source = e.source().to_string();
486                let err_string = e.to_string();
487                (
488                    e.ref_id(),
489                    pluginv2::DataResponse {
490                        frames: vec![],
491                        status,
492                        error: err_string,
493                        json_meta: vec![],
494                        error_source: source,
495                    },
496                )
497            }
498        })
499        .collect()
500        .await;
501        Ok(tonic::Response::new(pluginv2::QueryDataResponse {
502            responses,
503        }))
504    }
