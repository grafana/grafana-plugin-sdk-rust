6/// A no-op service which implements each of the available service traits.
7///
8/// This is used as a default type parameter of [`crate::backend::Plugin`]
9/// to enable a more type-safe API. The `Plugin` starts off in a state where
10/// every `service` field is assumed to be an optional `NoopService`;
11/// calling the `*_service()` methods constructs a new `Plugin` with type
12/// parameterized by the provided service, which replaces this struct.
13/// We can then use tonic's [`add_optional_service`][tonic::transport::server::Router::add_optional_service]
14/// to avoid actual having to construct this struct.
15///
16/// This struct cannot be constructed, so
17/// the `unreachable`s in this module should never be reached.
18#[derive(Debug)]
19pub struct NoopService {
20    _priv: (),
21}
22
23impl DataQueryError for Infallible {
