3/// The error returned when Grafana does not provide a requested configuration option.
4#[derive(Debug, Clone, thiserror::Error)]
5#[error("key {key} not found in Grafana config; a more recent version of Grafana may be required")]
6pub struct ConfigError {
7    key: String,
8}
9
10impl ConfigError {
11    fn from_key(key: String) -> Self {
12        Self { key }
13    }
14}
15
16type ConfigResult<T> = std::result::Result<T, ConfigError>;
17
18/// Configuration passed to the plugin from Grafana.
19#[derive(Clone)]
20pub struct GrafanaConfig {
21    config: HashMap<String, String>,
22}
23
24impl GrafanaConfig {
25    const APP_URL: &'static str = "GF_APP_URL";
26    const APP_CLIENT_SECRET: &'static str = "GF_PLUGIN_APP_CLIENT_SECRET";
27    // const CONCURRENT_QUERY_COUNT: &'static str = "GF_CONCURRENT_QUERY_COUNT";
28    // const USER_FACING_DEFAULT_ERROR: &'static str = "GF_USER_FACING_DEFAULT_ERROR";
29    // const SQL_ROW_LIMIT: &'static str = "GF_SQL_ROW_LIMIT";
30    // const SQL_MAX_OPEN_CONNS_DEFAULT: &'static str = "GF_SQL_MAX_OPEN_CONNS_DEFAULT";
31    // const SQL_MAX_IDLE_CONNS_DEFAULT: &'static str = "GF_SQL_MAX_IDLE_CONNS_DEFAULT";
32    // const SQL_MAX_CONN_LIFETIME_SECONDS_DEFAULT: &'static str =
33    //     "GF_SQL_MAX_CONN_LIFETIME_SECONDS_DEFAULT";
34    // const RESPONSE_LIMIT: &'static str = "GF_RESPONSE_LIMIT";
35
36    pub(crate) fn new(config: HashMap<String, String>) -> Self {
37        Self { config }
38    }
39
40    /// Get the value of a configuration option, if it was provided by Grafana.
41    fn get(&self, key: &str) -> ConfigResult<&String> {
42        self.config
43            .get(key)
44            .ok_or_else(|| ConfigError::from_key(key.to_string()))
45    }
46
47    /// Return the URL of the Grafana instance.
48    pub fn app_url(&self) -> ConfigResult<&String> {
49        self.get(Self::APP_URL)
50    }
51
52    /// Return the client secret for the app plugin's service account, if set.
53    ///
54    /// Plugins can request a service account be created by Grafana at startup
55    /// time by using the `iam` field of their `plugin.json` file. This method
56    /// will then return the client secret for that service account, which can
57    /// be used to authenticate with the Grafana API.
58    ///
59    /// See [this example plugin][example] for a full example of how to use this.
60    ///
61    /// [example]: https://github.com/grafana/grafana-plugin-examples/tree/main/examples/app-with-service-account
62    pub fn plugin_app_client_secret(&self) -> ConfigResult<&String> {
63        self.get(Self::APP_CLIENT_SECRET)
64    }
65}

67impl std::fmt::Debug for GrafanaConfig {
