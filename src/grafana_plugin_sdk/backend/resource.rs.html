20/// A request for a resource call.
21#[derive(Debug)]
22#[non_exhaustive]
23pub struct InnerCallResourceRequest<IS, JsonData, SecureJsonData>
24where
25    JsonData: fmt::Debug + DeserializeOwned,
26    SecureJsonData: DeserializeOwned,
27    IS: InstanceSettings<JsonData, SecureJsonData>,
28{
29    /// Details of the plugin instance from which the request originated.
30    pub plugin_context: PluginContext<IS, JsonData, SecureJsonData>,
31    /// The HTTP request.
32    pub request: Request<Bytes>,
33
34    _json_data: PhantomData<JsonData>,
35    _secure_json_data: PhantomData<SecureJsonData>,
36}
37
38impl<IS, JsonData, SecureJsonData> TryFrom<pluginv2::CallResourceRequest>
39    for InnerCallResourceRequest<IS, JsonData, SecureJsonData>
40where
41    JsonData: fmt::Debug + DeserializeOwned,
42    SecureJsonData: DeserializeOwned,
43    IS: InstanceSettings<JsonData, SecureJsonData>,
44{
45    type Error = ConvertFromError;
46    fn try_from(other: pluginv2::CallResourceRequest) -> Result<Self, Self::Error> {
47        let mut request = Request::builder();
48        {
49            let headers = request
50                .headers_mut()
51                .expect("Request should always be valid at this point");
52            for (k, vs) in other.headers.iter() {
53                let header_name =
54                    k.parse::<HeaderName>()
55                        .map_err(|source| ConvertFromError::InvalidRequest {
56                            source: source.into(),
57                        })?;
58                for v in &vs.values {
59                    let header_val = v.parse::<HeaderValue>().map_err(|source| {
60                        ConvertFromError::InvalidRequest {
61                            source: source.into(),
62                        }
63                    })?;
64                    headers.append(header_name.clone(), header_val);
65                }
66            }
67        }
68        Ok(Self {
69            plugin_context: other
70                .plugin_context
71                .ok_or(ConvertFromError::MissingPluginContext)
72                .and_then(TryInto::try_into)?,
73            request: request
74                .method(other.method.as_str())
75                .uri(format!("/{}", other.url))
76                .body(other.body)
77                .map_err(|source| ConvertFromError::InvalidRequest { source })?,
78            _json_data: PhantomData,
79            _secure_json_data: PhantomData,
80        })
81    }
82}

84impl TryFrom<Response<Bytes>> for pluginv2::CallResourceResponse {
85    type Error = ConvertToError;
86    fn try_from(mut other: Response<Bytes>) -> Result<Self, Self::Error> {
87        let grouped_headers = other.headers_mut().drain().chunk_by(|x| x.0.clone());
88        let headers = grouped_headers
89            .into_iter()
90            .map(|(k, values)| {
91                Ok((
92                    k.as_ref()
93                        .map(|h| h.to_string())
94                        .ok_or(ConvertToError::InvalidResponse)?,
95                    pluginv2::StringList {
96                        values: values
97                            .map(|v| {
98                                v.1.to_str()
99                                    .map(|s| s.to_string())
100                                    .map_err(|_| ConvertToError::InvalidResponse)
101                            })
102                            .collect::<Result<_, _>>()?,
103                    },
104                ))
105            })
106            .collect::<Result<_, _>>()?;
107        drop(grouped_headers);
108        Ok(Self {
109            code: other.status().as_u16().into(),
110            headers,
111            body: other.into_body(),
112        })
113    }
114}

133    pluginv2::CallResourceResponse {
134        code: 200,
135        headers: std::collections::HashMap::new(),

147/// Trait for plugins that can handle arbitrary resource requests.
148///
149/// Implementing this trait allows plugins to handle a wide variety of use cases beyond
150/// 'just' responding to requests for data and returning dataframes.
151///
152/// See <https://grafana.com/docs/grafana/latest/developers/plugins/backend/#resources> for
153/// some examples of how this can be used.
154///
155/// # Examples
156///
157/// ## Stateful service
158///
159/// The following shows an example implementation of [`ResourceService`] which handles
160/// two endpoints:
161/// - /echo, which echos back the request's URL, headers and body in three responses,
162/// - /count, which increments the plugin's internal count and returns it in a response.
163///
164/// ```rust
165/// use std::sync::{atomic::{AtomicUsize, Ordering}, Arc};
166///
167/// use async_stream::stream;
168/// use bytes::Bytes;
169/// use grafana_plugin_sdk::{backend, prelude::*};
170/// use http::Response;
171/// use thiserror::Error;
172///
173/// #[derive(Clone, Debug, GrafanaPlugin)]
174/// #[grafana_plugin(plugin_type = "app")]
175/// struct Plugin(Arc<AtomicUsize>);
176///
177/// impl Plugin {
178///     // Increment the counter and return the stringified result in a `Response`.
179///     fn inc_and_respond(&self) -> Response<Bytes> {
180///         Response::new(
181///             self.0
182///                 .fetch_add(1, Ordering::SeqCst)
183///                 .to_string()
184///                 .into_bytes()
185///                 .into()
186///         )
187///     }
188/// }
189///
190/// #[derive(Debug, Error)]
191/// enum ResourceError {
192///     #[error("HTTP error: {0}")]
193///     Http(#[from] http::Error),
194///
195///     #[error("Path not found")]
196///     NotFound,
197/// }
198///
199/// impl backend::ErrIntoHttpResponse for ResourceError {}
200///
201/// #[backend::async_trait]
202/// impl backend::ResourceService for Plugin {
203///     type Error = ResourceError;
204///     type InitialResponse = Response<Bytes>;
205///     type Stream = backend::BoxResourceStream<Self::Error>;
206///     async fn call_resource(&self, r: backend::CallResourceRequest<Self>) -> Result<(Self::InitialResponse, Self::Stream), Self::Error> {
207///         let count = Arc::clone(&self.0);
208///         let response_and_stream = match r.request.uri().path() {
209///             // Just send back a single response.
210///             "/echo" => Ok((
211///                 Response::new(r.request.into_body().into()),
212///                 Box::pin(futures::stream::empty()) as Self::Stream,
213///             )),
214///             // Send an initial response with the current count, then stream the gradually
215///             // incrementing count back to the client.
216///             "/count" => Ok((
217///                 Response::new(
218///                     count
219///                         .fetch_add(1, Ordering::SeqCst)
220///                         .to_string()
221///                         .into_bytes()
222///                         .into(),
223///                 ),
224///                 Box::pin(async_stream::try_stream! {
225///                     loop {
226///                         let body = count
227///                             .fetch_add(1, Ordering::SeqCst)
228///                             .to_string()
229///                             .into_bytes()
230///                             .into();
231///                         yield body;
232///                     }
233///                 }) as Self::Stream,
234///             )),
235///             _ => return Err(ResourceError::NotFound),
236///         };
237///         response_and_stream
238///     }
239/// }
240/// ```
241#[tonic::async_trait]
242pub trait ResourceService: GrafanaPlugin {
243    /// The error type that can be returned by individual responses.
244    type Error: std::error::Error + ErrIntoHttpResponse + Send;
245
246    /// The type returned as the initial response returned back to Grafana.
247    ///
248    /// This must be convertable into a `http::Response<Bytes>`.
249    type InitialResponse: IntoHttpResponse + Send;
250
251    /// The type of stream of optional additional data returned by `run_stream`.
252    ///
253    /// This will generally be impossible to name directly, so returning the
254    /// [`BoxResourceStream`] type alias will probably be more convenient.
255    type Stream: futures_core::Stream<Item = Result<Bytes, Self::Error>> + Send;
256
257    /// Handle a resource request.
258    ///
259    /// It is completely up to the implementor how to handle the incoming request.
260    ///
261    /// A stream of responses can be returned. A simple way to return just a single response
262    /// is to use [`futures_util::stream::once`].
263    async fn call_resource(
264        &self,
265        request: CallResourceRequest<Self>,
266    ) -> Result<(Self::InitialResponse, Self::Stream), Self::Error>;
267}

269#[tonic::async_trait]
270impl<T> pluginv2::resource_server::Resource for T
271where
272    T: ResourceService + Send + Sync + 'static,
273{
274    type CallResourceStream = Pin<
275        Box<
276            dyn futures_core::Stream<Item = Result<pluginv2::CallResourceResponse, tonic::Status>>
277                + Send,
278        >,
279    >;
280
281    #[tracing::instrument(skip(self), level = "debug")]
282    async fn call_resource(
283        &self,
284        request: tonic::Request<pluginv2::CallResourceRequest>,
285    ) -> Result<tonic::Response<Self::CallResourceStream>, tonic::Status> {
286        let request = request
287            .into_inner()
288            .try_into()
289            .map_err(ConvertFromError::into_tonic_status)?;
290        let (initial_response, stream) = match ResourceService::call_resource(self, request)
291            .await
292            .map_err(|e: <Self as ResourceService>::Error| e.into_http_response())
293        {
294            Ok(x) => x,
295            Err(resp) => {
296                let response = resp.map_err(|e| tonic::Status::internal(e.to_string()))?;
297                let response = pluginv2::CallResourceResponse::try_from(response)
298                    .map_err(|e| tonic::Status::internal(e.to_string()));
299                let stream = futures_util::stream::once(futures_util::future::ready(response));
300                return Ok(tonic::Response::new(Box::pin(stream)));
301            }
302        };
303        let initial_response_converted = initial_response
304            .into_http_response()
305            .await
306            .map_err(|e| tonic::Status::internal(e.to_string()))
307            .and_then(|response| {
308                pluginv2::CallResourceResponse::try_from(response)
309                    .map_err(|e| tonic::Status::internal(e.to_string()))
310            });
311        let stream =
312            futures_util::stream::once(futures_util::future::ready(initial_response_converted))
313                .chain(stream.map(|response| {
314                    response
315                        .map(body_to_response)
316                        .map_err(|e| tonic::Status::internal(e.to_string()))
317                }));
318        Ok(tonic::Response::new(Box::pin(stream)))
319    }
320}

322/// Trait indicating that a type can be fallibly converted into a `http::Response<Bytes>`.
323///
324/// This is a separate trait (rather than using `TryFrom`/`TryInto`) because it may
325/// need to be async.
326#[tonic::async_trait]
327pub trait IntoHttpResponse {
328    /// Performs the conversion.
329    async fn into_http_response(self) -> Result<http::Response<Bytes>, Box<dyn std::error::Error>>;
330}

332#[tonic::async_trait]
333impl IntoHttpResponse for http::Response<Bytes> {
334    async fn into_http_response(self) -> Result<http::Response<Bytes>, Box<dyn std::error::Error>> {
335        Ok(self)
336    }
337}

368/// Trait describing how an error should be converted into a `http::Response<Bytes>`.
369pub trait ErrIntoHttpResponse: std::error::Error + Sized {
370    /// Convert this error into a HTTP response.
371    ///
372    /// The default implementation returns a response with status code 500 (Internal Server Error)
373    /// and the `Display` implementation of `Self` inside the `"error"` field of a JSON object
374    /// in the body.
375    ///
376    /// Implementors may wish to override this if they wish to provide an alternative status code
377    /// depending on, for example, the type of error returned from a resource call.
378    ///
379    /// # Example
380    ///
381    /// ```rust
382    /// use bytes::Bytes;
383    /// use grafana_plugin_sdk::backend;
384    /// use thiserror::Error;
385    ///
386    /// #[derive(Debug, Error)]
387    /// enum ResourceError {
388    ///     #[error("HTTP error: {0}")]
389    ///     Http(#[from] http::Error),
390    ///
391    ///     #[error("Path not found")]
392    ///     NotFound,
393    /// }
394    ///
395    /// impl backend::ErrIntoHttpResponse for ResourceError {
396    ///     fn into_http_response(self) -> Result<http::Response<Bytes>, Box<dyn std::error::Error>> {
397    ///         let status = match &self {
398    ///             Self::Http(_) => http::StatusCode::INTERNAL_SERVER_ERROR,
399    ///             Self::NotFound => http::StatusCode::NOT_FOUND,
400    ///         };
401    ///         Ok(http::Response::builder()
402    ///             .status(status)
403    ///             .body(Bytes::from(serde_json::to_vec(
404    ///                 &serde_json::json!({"error": self.to_string()}),
405    ///             )?))?)
406    ///     }
407    /// }
408    /// ```
409    fn into_http_response(self) -> Result<http::Response<Bytes>, Box<dyn std::error::Error>> {
410        Ok(Response::builder()
411            .status(StatusCode::INTERNAL_SERVER_ERROR)
412            .body(Bytes::from(serde_json::to_vec(
413                &serde_json::json!({"error": self.to_string()}),
414            )?))?)
415    }
416}
