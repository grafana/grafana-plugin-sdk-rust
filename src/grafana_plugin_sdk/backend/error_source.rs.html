1use std::fmt;
2
3/// The source of an error.
4///
5/// This is used to indicate whether the error occurred in the plugin or downstream.
6#[derive(Debug, Default, Clone, Copy, PartialEq, Eq, Hash)]
7#[non_exhaustive]
8pub enum ErrorSource {
9    /// The error occurred in the plugin.
10    #[default]
11    Plugin,
12    /// The error occurred downstream of the plugin.
13    Downstream,
14}
15
16impl fmt::Display for ErrorSource {
17    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
18        match self {
19            Self::Plugin => f.write_str("plugin"),
20            Self::Downstream => f.write_str("downstream"),
21        }
22    }
23}
