1/*! Functionality for use by backend plugins.
2
3Backend plugins are executables that expose a gRPC server, which Grafana uses to
4communicate all information. This SDK uses [tonic] as its gRPC implementation.
5
6The basic requirements for a backend plugin are to provide a `main` function which:
7
8- first, calls the [`initialize`] function from this module to initialize the plugin. This **must**
9  be done before any other code can output to stdout or stderr!
10- performs any setup required by the plugin (such as connecting to databases or initializing any state)
11- creates the struct representing the plugin's services, which should implement one of the various traits exposed
12  by this module
13- create a [`Plugin`] to manage the plugin's lifecycle with Grafana
14- use the `Plugin::*_service` methods to attach your plugin
15- begin serving on the listener returned by `initialize` using [`Plugin::start`].
16
17If you're not sure where to start, take a look at the [`Plugin`] struct, its four important methods, and
18the trait bounds for each of them - those give an idea of what your plugin will need to implement.
19
20# Logging and `tracing`
21
22The SDK provides a preconfigured [`tracing_subscriber::fmt::Layer`] which, when installed,
23will emit structured logs in a format understood by Grafana. Use the [`layer`] function
24to get the `Layer`, and install it using `tracing_subscriber::Registry::with`. Alternatively
25use [`Plugin::init_subscriber`] to automatically install a subscriber using the `RUST_LOG`
26environment variable to set the directive (defaulting to `info` if not set).
27
28Once the layer is installed, any logs emitted by the various [`log`](https://docs.rs/log) or [`tracing`] macros
29will be emitted to Grafana.
30
31# Example
32
33```rust,no_run
34use futures_util::stream::FuturesOrdered;
35use grafana_plugin_sdk::{backend, data, prelude::*};
36use thiserror::Error;
37use tonic::transport::Server;
38use tracing::info;

40#[derive(Clone, Debug, GrafanaPlugin)]
41#[grafana_plugin(plugin_type = "datasource")]
42struct MyPlugin;
43
44/// An error that may occur during a query.
45///
46/// This must store the `ref_id` of the query so that Grafana can line it up.
47#[derive(Debug, Error)]
48#[error("Error querying backend for query {ref_id}: {source}")]
49struct QueryError {
50    source: data::Error,
51    ref_id: String,
52}
53
54impl backend::DataQueryError for QueryError {
55    fn ref_id(self) -> String {
56        self.ref_id
57    }
58}
59
60#[tonic::async_trait]
61impl backend::DataService for MyPlugin {
62
63    /// The type of JSON data sent from Grafana to our backend plugin.
64    ///
65    /// This will correspond to the `TQuery` type parameter of the frontend
66    /// datasource.
67    ///
68    /// We can use `serde_json::Value` if we want to accept any JSON.
69    type Query = serde_json::Value;
70
71    /// The type of error that could be returned by an individual query.
72    type QueryError = QueryError;
73
74    /// The type of iterator we're returning.
75    ///
76    /// In general the concrete type will be impossible to name in advance,
77    /// so the `backend::BoxDataResponseStream` type alias will be useful.
78    type Stream = backend::BoxDataResponseStream<Self::QueryError>;
79
80    /// Respond to a request for data from Grafana.
81    ///
82    /// This request will contain zero or more queries, as well as information
83    /// about the datasource instance on behalf of which this request is made,
84    /// such as address, credentials, etc.
85    ///
86    /// Our plugin must respond to each query and return an iterator of `DataResponse`s,
87    /// which themselves can contain zero or more `Frame`s.
88    async fn query_data(&self, request: backend::QueryDataRequest<Self::Query, Self>) -> Self::Stream {
89        Box::pin(
90            request
91                .queries
92                .into_iter()
93                .map(|x| async {
94                    // Here we create a single response Frame for each query.
95                    // Frames can be created from iterators of fields using [`IntoFrame`].
96                    Ok(backend::DataResponse::new(
97                        x.ref_id.clone(),
98                        // Return zero or more frames.
99                        // A real implementation would fetch this data from a database
100                        // or something.
101                        vec![[
102                            [1_u32, 2, 3].into_field("x"),
103                            ["a", "b", "c"].into_field("y"),
104                        ]
105                        .into_frame("foo")
106                        .check()
107                        .map_err(|source| QueryError {
108                            ref_id: x.ref_id,
109                            source,
110                        })?],
111                    ))
112                })
113                .collect::<FuturesOrdered<_>>(),
114        )
115    }
116}
117
118#[grafana_plugin_sdk::main(services(data))]
119async fn plugin() -> MyPlugin {
120    // Create our plugin struct. Any state, such as a database connection, should be
121    // held here, perhaps inside an `Arc` if required.
122    MyPlugin
123}
124```

194impl ShutdownHandler {
195    #[must_use]
196    fn new(address: SocketAddr) -> Self {
197        Self { address }
198    }
199
200    fn spawn(self) -> impl std::future::Future<Output = ()> {
201        tokio::spawn(async move {
202            let listener = TcpListener::bind(&self.address).await.map_err(|e| {
203                tracing::warn!("Error creating shutdown handler: {}", e);
204                e
205            })?;
206            tracing::debug!(address = %self.address, "Shutdown handler started on {}", &self.address);
207            Ok::<_, std::io::Error>(listener.accept().await.map(|_| ()))
208        })
209        .map(|_| ())
210    }
211}

217/// Main entrypoint into the Grafana plugin SDK.
218///
219/// A `Plugin` handles the negotiation with Grafana, adding gRPC health checks,
220/// serving the various `Service`s, and gracefully exiting if configured.
221///
222/// # Shutdown Handler
223///
224/// A plugin can be configured to gracefully shutdown by calling the [`Plugin::shutdown_handler`]
225/// method. This will spawn a new task which simply waits for _any_ TCP connection on
226/// the given address, and triggers a graceful shutdown of the underlying gRPC server.
227/// This is helpful during development using Docker: a tool such as [`cargo-watch`][cargo-watch]
228/// can be used to rebuild on changes, then use `nc` to trigger a shutdown of the plugin. Grafana
229/// will automatically plugins when their process exits, so this avoids having to restart the
230/// Grafana server on every change. An example configuration for this can be seen in the
231/// [grafana-sample-backend-plugin-rust] repository.
232///
233/// [cargo-watch]: https://crates.io/crates/cargo-watch
234/// [grafana-sample-backend-plugin-rust]: https://github.com/sd2k/grafana-sample-backend-plugin-rust
235///
236/// # Type parameters
237///
238/// The four type parameters of a `Plugin` represent the services that this plugin
239/// is configured to run. When created using [`Plugin::new`] the plugin has the
240/// `NoopService` associated with each type, which simply does not register anything
241/// for this piece of functionality. Calling the various methods on `Plugin` (such
242/// as [`Plugin::data_service`] and [`Plugin::stream_service`]) will return a new
243/// `Plugin` with your plugin implementation registered for that functionality.
244///
245/// The type parameters stand for:
246///
247/// - `D`, the diagnostic service
248/// - `Q`, the data service ('Q' stands for 'query', here)
249/// - `R`, the resource service
250/// - `S`, the streaming service
251///
252/// # Example:
253///
254/// ```rust
255/// use futures_util::stream::FuturesOrdered;
256/// use grafana_plugin_sdk::{backend, prelude::*};
257/// use thiserror::Error;
258///
259/// #[derive(Clone, GrafanaPlugin)]
260/// #[grafana_plugin(plugin_type = "datasource")]
261/// struct MyPlugin;
262///
263/// /// An error that may occur during a query.
264/// ///
265/// /// This must store the `ref_id` of the query so that Grafana can line it up.
266/// #[derive(Debug, Error)]
267/// #[error("Error querying backend for {ref_id}")]
268/// struct QueryError {
269///     ref_id: String,
270/// };
271///
272/// impl backend::DataQueryError for QueryError {
273///     fn ref_id(self) -> String {
274///         self.ref_id
275///     }
276/// }
277///
278/// #[tonic::async_trait]
279/// impl backend::DataService for MyPlugin {
280///
281///     /// The type of JSON data sent from Grafana to our backend plugin.
282///     ///
283///     /// This will correspond to the `TQuery` type parameter of the frontend
284///     /// datasource.
285///     ///
286///     /// We can use `serde_json::Value` if we want to accept any JSON.
287///     type Query = serde_json::Value;
288///
289///     /// The type of error that could be returned by an individual query.
290///     type QueryError = QueryError;
291///
292///     /// The type of iterator we're returning.
293///     ///
294///     /// In general the concrete type will be impossible to name in advance,
295///     /// so the `backend::BoxDataResponseStream` type alias will be useful.
296///     type Stream = backend::BoxDataResponseStream<Self::QueryError>;
297///
298///     /// Respond to a request for data from Grafana.
299///     ///
300///     /// This request will contain zero or more queries, as well as information
301///     /// about the datasource instance on behalf of which this request is made,
302///     /// such as address, credentials, etc.
303///     ///
304///     /// Our plugin must respond to each query and return an iterator of `DataResponse`s,
305///     /// which themselves can contain zero or more `Frame`s.
306///     async fn query_data(&self, request: backend::QueryDataRequest<Self::Query, Self>) -> Self::Stream {
307///         Box::pin(
308///             request
309///                 .queries
310///                 .into_iter()
311///                 .map(|x| async {
312///                     // Here we create a single response Frame for each query.
313///                     // Frames can be created from iterators of fields using [`IntoFrame`].
314///                     Ok(backend::DataResponse::new(
315///                         x.ref_id.clone(),
316///                         // Return zero or more frames.
317///                         // A real implementation would fetch this data from a database
318///                         // or something.
319///                         vec![[
320///                             [1_u32, 2, 3].into_field("x"),
321///                             ["a", "b", "c"].into_field("y"),
322///                         ]
323///                         .into_frame("foo")
324///                         .check()
325///                         .map_err(|source| QueryError {
326///                             ref_id: x.ref_id,
327///                         })?],
328///                     ))
329///                 })
330///                 .collect::<FuturesOrdered<_>>(),
331///         )
332///     }
333/// }
334///
335/// #[grafana_plugin_sdk::main(
336///     services(data),
337///     shutdown_handler = "0.0.0.0:10001",
338/// )]
339/// async fn plugin() -> MyPlugin {
340///     MyPlugin
341/// }
342/// ```

479impl<D, Q, R, S> Plugin<D, Q, R, S>
480where
481    D: DiagnosticsService + Send + Sync + 'static,
482    Q: DataService + Send + Sync + 'static,
483    R: ResourceService + Send + Sync + 'static,
484    S: StreamService + Send + Sync + 'static,
485{
486    /// Start the plugin.
487    ///
488    /// This adds all of the configured services, spawns a shutdown handler
489    /// (if configured), and blocks while the plugin runs.
490    ///
491    /// # Panics
492    ///
493    /// This will panic if `init_subscriber(true)` has been set and another
494    /// global subscriber has already been installed. If you are initializing your
495    /// own `Subscriber`, you should instead use the [`layer`] function to add a
496    /// Grafana-compatible `tokio_subscriber::fmt::Layer` to your subscriber.
497    pub async fn start(self, listener: TcpListener) -> Result<(), Error> {
498        if self.init_subscriber {
499            let filter =
500                EnvFilter::try_from_default_env().unwrap_or_else(|_| EnvFilter::new("info"));
501            tracing_subscriber::registry()
502                .with(layer())
503                .with(filter)
504                .init();
505        }
506        let (health_reporter, health_service) = tonic_health::server::health_reporter();
507        let mut plugins = Vec::with_capacity(5);
508        if self.diagnostics_service.is_some() {
509            health_reporter.set_serving::<DiagnosticsServer<D>>().await;
510            plugins.push("diagnostics");
511        }
512        if self.data_service.is_some() {
513            health_reporter.set_serving::<DataServer<Q>>().await;
514            plugins.push("data");
515        }
516        if self.resource_service.is_some() {
517            health_reporter.set_serving::<ResourceServer<R>>().await;
518            plugins.push("resource");
519        }
520        if self.stream_service.is_some() {
521            health_reporter.set_serving::<StreamServer<S>>().await;
522            plugins.push("stream");
523        }
524        // Log the services included in the plugin, identically to the Go SDK.
525        tracing::debug!(
526            plugins = %format!("[{}]", plugins.join(" ")),
527            "Serving plugin"
528        );
529        let router = tonic::transport::Server::builder()
530            .trace_fn(|_| tracing::debug_span!("grafana-plugin-sdk"))
531            .add_service(health_service)
532            .add_optional_service(self.diagnostics_service.map(DiagnosticsServer::new))
533            .add_optional_service(self.data_service.map(DataServer::new))
534            .add_optional_service(self.resource_service.map(ResourceServer::new))
535            .add_optional_service(self.stream_service.map(StreamServer::new));
536        if let Some(handler) = self.shutdown_handler {
537            let handler = handler.spawn();
538            router
539                .serve_with_incoming_shutdown(TcpListenerStream::new(listener), handler.map(|_| ()))
540                .await?;
541        } else {
542            router
543                .serve_with_incoming(TcpListenerStream::new(listener))
544                .await?;
545        }
546        Ok(())
547    }
548}

571/// Create a `tracing` [`Layer`][tracing_subscriber::Layer] configured to log events in a format understood by Grafana.
572///
573/// The returned layer should be installed into the tracing subscriber registry, with an optional env filter.
574///
575/// # Example
576///
577/// Installing the layer with the default `EnvFilter` (using the `RUST_LOG` environment variable):
578///
579/// ```rust
580/// use grafana_plugin_sdk::backend;
581/// use tracing_subscriber::{prelude::*, EnvFilter};
582///
583/// tracing_subscriber::registry()
584///     .with(backend::layer())
585///     .with(EnvFilter::from_default_env())
586///     .init();
587/// ```
588pub fn layer<S: tracing::Subscriber + for<'a> LookupSpan<'a>>(

590    tracing_subscriber::fmt::layer()
591        .with_timer(UtcTime::new(HCLOG_TIME_FORMAT))
592        .with_writer(io::stderr as fn() -> std::io::Stderr)
593        .event_format(tracing_fmt::HCLogJson::default())
594        .fmt_fields(JsonFields::new())
595}

612/// Errors occurring when trying to interpret data passed from Grafana to this SDK.
613///
614/// Generally any errors should be considered a bug and should be reported.
615#[serde_as]
616#[derive(Debug, Error, Serialize)]
617#[serde(rename_all = "camelCase", tag = "type")]
618#[non_exhaustive]
619pub enum ConvertFromError {
620    /// The `time_range` was missing from the query.
621    #[error("time_range missing from query")]
622    MissingTimeRange,
623    /// The timestamp passed from Grafana was invalid.
624    #[error("error interpreting timestamp ({timestamp}) from Grafana")]
625    InvalidTimestamp {
626        /// The original timestamp in milliseconds.
627        timestamp: i64,
628    },
629    /// The `plugin_context` was missing from the request.
630    #[error("plugin_context missing from request")]
631    MissingPluginContext,
632    /// The query's JSON data sent by Grafana could not be deserialized.
633    #[error("unexpected query JSON (got {json}): {err}")]
634    UnexpectedQueryJson {
635        /// The underlying JSON error.
636        #[serde_as(as = "DisplayFromStr")]
637        err: serde_json::Error,
638        /// The JSON for which deserialization was attempted.
639        json: String,
640    },
641    /// The plugin instance's JSON data sent by Grafana could not be deserialized
642    /// into the backend representation.
643    #[error("unexpected plugin JSON data (got {json}): {err}")]
644    UnexpectedJsonData {
645        /// The underlying JSON error.
646        #[serde_as(as = "DisplayFromStr")]
647        err: serde_json::Error,
648        /// The JSON for which deserialization was attempted.
649        json: String,
650    },
651    /// The plugin instance's 'secure' JSON data sent by Grafana could not be converted into
652    /// the backend representation.
653    #[error(
654        "unexpected plugin secure JSON data (got keys: [{keys}]): {err}",
655        keys = secure_json_keys.join(", ")
656    )]
657    UnexpectedSecureJson {
658        /// The underlying JSON error.
659        #[serde_as(as = "DisplayFromStr")]
660        err: serde_json::Error,
661        /// The keys found in the underlying decrypted secure JSON.
662        ///
663        /// Values are not shown because they are likely to be secret.
664        secure_json_keys: Vec<String>,
665    },
666    /// The frame provided by Grafana was malformed.
667    #[error("invalid frame: {source}")]
668    InvalidFrame {
669        /// The underlying JSON error.
670        #[serde_as(as = "DisplayFromStr")]
671        source: serde_json::Error,
672    },
673    /// The resource request was not a valid HTTP request.
674    #[error("invalid HTTP request: {source}")]
675    InvalidRequest {
676        /// The underlying `http` error.
677        #[serde_as(as = "DisplayFromStr")]
678        source: http::Error,
679    },
680    /// The role string provided by Grafana didn't match the roles known by the SDK.
681    #[error("Unknown role: {0}")]
682    UnknownRole(String),
683    /// The path provided by Grafana was invalid.
684    #[error("Invalid path: {source}")]
685    InvalidPath {
686        /// The underlying reason for the error.
687        #[from]
688        source: live::ChannelError,
689    },
690}

694        Self::invalid_argument(other.to_string())
695    }

704/// Errors occurring when trying to convert data into a format understood by Grafana.
705///
706/// If these errors occur it probably means something invalid was passed to the SDK
707/// somehow, such as a `HashMap` with numeric keys passed as a JSON value.
708#[derive(Debug, Error)]
709#[non_exhaustive]
710pub enum ConvertToError {
711    /// The resource response was not a valid HTTP response.
712    #[error("invalid HTTP response")]
713    InvalidResponse,
714    /// The JSON provided was invalid.
715    #[error("invalid JSON: {err}")]
716    InvalidJson {
717        /// The underlying JSON error.
718        err: serde_json::Error,
719    },
720    /// The frame provided could not be serialized.
721    #[error("invalid frame: {source}")]
722    InvalidFrame {
723        /// The underlying JSON error.
724        source: serde_json::Error,
725    },
726}

728type ConvertFromResult<T> = std::result::Result<T, ConvertFromError>;
729
730fn read_json<T>(jdoc: &[u8]) -> Result<T, serde_json::Error>
731where
732    T: DeserializeOwned,
733{
734    // Grafana sometimes sends an empty string instead of an empty map, probably
735    // because of some zero value Golang stuff?
736    let jdoc = jdoc.is_empty().then(|| b"{}".as_slice()).unwrap_or(jdoc);
737    serde_json::from_slice(jdoc)
738}
739
740fn read_json_query<T>(jdoc: &[u8]) -> ConvertFromResult<T>
741where
742    T: DeserializeOwned,
743{
744    read_json(jdoc).map_err(|err| ConvertFromError::UnexpectedQueryJson {
745        err,
746        json: String::from_utf8(jdoc.to_vec())
747            .unwrap_or_else(|_| format!("non-utf8 string: {}", String::from_utf8_lossy(jdoc))),
748    })
749}
750
751fn read_json_data<T>(jdoc: &[u8]) -> ConvertFromResult<T>
752where
753    T: DeserializeOwned,
754{
755    read_json(jdoc).map_err(|err| ConvertFromError::UnexpectedJsonData {
756        err,
757        json: String::from_utf8(jdoc.to_vec())
758            .unwrap_or_else(|_| format!("non-utf8 string: {}", String::from_utf8_lossy(jdoc))),
759    })
760}

837fn convert_secure_json_data<T: DeserializeOwned>(
838    secure_json: &HashMap<String, String>,
839) -> Result<T, ConvertFromError> {
840    serde_json::from_value(Value::Object(serde_json::Map::from_iter(
841        secure_json
842            .iter()
843            .map(|(k, v)| (k.clone(), Value::String(v.clone()))),
844    )))
845    .map_err(|e| ConvertFromError::UnexpectedSecureJson {
846        err: e,
847        secure_json_keys: secure_json.keys().cloned().collect(),
848    })
849}
